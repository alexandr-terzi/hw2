false && true || true // true, потому что между фолс И тру, И возвращает первое ложное значение, затем между ИЛИ - истинное

false && (true || true) // false, сначала действие в скобках, там тру, затем между фолс И тру первое ложное значение возвращается

+'40' + +'2' // 42, потому что унарный плюс превращает строки в числа

'2' + 3 ** 2 // '29', приоритетность степени выше, затем складываем строки

3 ** 2 / 3 // 3

3 ** (9 / 3) // 27, сначала действие в скобках

'10' + (5 === 6) // '100', 5 === 6 - это фолс, фолс преобразуется в числовое значение - 0, затем складываем строки

'10' - 5 === 6 // false, строка преобразуется в число

undefined + 1 // NaN, undefined становится NaN после числового преобразования

(+null == false) < 1 // false, в скобках получаем 0 == 0 - тру, соответственно 1 не меньше 1

(+null == false) < 1 ** 5 // false, в скобках получаем 0 == 0 - тру, соответственно 1 не меньше 1

var number = 5;

2 * ++number // 12, префиксная форма увеличивает и возвращает новое значение, т.е. 2 * 6 = 12

2 * number--; // 10, постфиксная форма уменьшает, но возвращает старое значение, т.е. 2 * 5 = 10

console.log(number) // 5
